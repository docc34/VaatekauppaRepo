CREATE TABLE `post` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `label` varchar(45) NOT NULL,
  `price` double NOT NULL,
  `description` varchar(400) NOT NULL,
  `userId` int(11) NOT NULL,
  `taxId` int(11) NOT NULL,
  `discount` int(11) DEFAULT NULL,
  `material` varchar(45) DEFAULT NULL,
  `length` varchar(60) DEFAULT NULL,
  `sleeveLength` varchar(60) DEFAULT NULL,
  `colorId` int(11) DEFAULT NULL,
  `sizeId` int(11) DEFAULT NULL,
  `deliveryDaysEstimateStarting` int(11) DEFAULT NULL,
  `deliveryDaysEstimateEnding` int(11) DEFAULT NULL,
  `deleted` tinyint(4) NOT NULL,
  PRIMARY KEY (`id`),
  KEY `fk_jobPost_userdata1_idx` (`userId`),
  KEY `fk_post_color2_idx` (`colorId`),
  KEY `fk_post_size1_idx` (`sizeId`),
  KEY `fk_post_tax1_idx` (`taxId`),
  CONSTRAINT `fk_jobPost_userdata1` FOREIGN KEY (`userId`) REFERENCES `user` (`id`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `fk_post_color2` FOREIGN KEY (`colorId`) REFERENCES `color` (`id`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `fk_post_size1` FOREIGN KEY (`sizeId`) REFERENCES `size` (`id`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `fk_post_tax1` FOREIGN KEY (`taxId`) REFERENCES `tax` (`id`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8
