CREATE TABLE `ratings` (
  `ratingsId` int(11) NOT NULL AUTO_INCREMENT,
  `postId` int(11) NOT NULL,
  `userId` int(11) NOT NULL,
  `stars` double NOT NULL,
  `review` varchar(200) DEFAULT NULL,
  PRIMARY KEY (`ratingsId`),
  KEY `fk_ratings_post1_idx` (`postId`),
  KEY `fk_ratings_profile1_idx` (`userId`),
  CONSTRAINT `fk_ratings_post1` FOREIGN KEY (`postId`) REFERENCES `post` (`postId`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `fk_ratings_profile1` FOREIGN KEY (`userId`) REFERENCES `profile` (`userId`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8
